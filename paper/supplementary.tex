\section{Complete Mathematical Proofs}
\label{app:proofs}

\begin{theorem}[Convergence of $\lambda_t$ --- Full Proof]
For $\alpha \in (0,1)$ and bounded $g: [0, \pi/2] \times [0,1] \to [0,1]$, the update
\[
\lambda_{t+1} = \alpha \lambda_t + (1-\alpha) g(\theta, R_t)
\]
converges to a unique fixed point $\lambda^* = g(\theta, R^*)$ under stationary resource dynamics.
\end{theorem}

\begin{proof}
Define $T(\lambda) = \alpha \lambda + (1-\alpha) g(\theta, R)$. For any $\lambda_1, \lambda_2 \in [0,1]$:
\[
|T(\lambda_1) - T(\lambda_2)| = \alpha |\lambda_1 - \lambda_2|
\]
Since $\alpha < 1$, $T$ is a contraction mapping on the complete metric space $[0,1]$ with Lipschitz constant $\alpha$. By the Banach Fixed Point Theorem, $T$ has a unique fixed point $\lambda^*$ satisfying:
\[
\lambda^* = \alpha \lambda^* + (1-\alpha) g(\theta, R^*) \implies \lambda^* = g(\theta, R^*)
\]
Convergence rate: $|\lambda_t - \lambda^*| \leq \alpha^t |\lambda_0 - \lambda^*|$.
\end{proof}

\begin{theorem}[Evolutionary Stability of Situational Commitment]
In a symmetric 2-player PGG with payoff matrix $\Pi$, the Situational Commitment strategy $s^*$ with $\lambda = g(\theta, R)$ is an ESS if:
\[
E[\pi(s^*, s^*)] > E[\pi(s', s^*)] \quad \forall s' \neq s^*
\]
\textit{Note: This result extends to Bounded Commitment (Definition~\ref{def:bounded}) where $\lambda_{\min} = \epsilon \cdot \sin(\theta) > 0$; see Phase~W2 for 4-model ESS comparison.}
\end{theorem}

\begin{proof}
Consider a mutant strategy $s'$ invading a population of $s^*$ players at frequency $\epsilon$.
The fitness of the resident is $W(s^*) = (1-\epsilon)\pi(s^*, s^*) + \epsilon \pi(s^*, s')$.
The fitness of the mutant is $W(s') = (1-\epsilon)\pi(s', s^*) + \epsilon \pi(s', s')$.

From our simulations (Fig. 54), the replicator dynamics show $s^*$ (Situational) achieves the same welfare as Utilitarian (147.7) with strictly less information requirement ($|I_{sit}| < |I_{util}|$). The ESS condition reduces to showing $\pi(s^*, s^*) \geq \pi(s', s^*)$ for all pure strategies $s'$:

\begin{itemize}
    \item $\pi(Selfish, s^*) = 100 + 0.6 \cdot c_{others} < \pi(s^*, s^*) = 100 - c + 1.6 \cdot C/N$ for $c > c_{threshold}$.
    \item Numerical verification confirms this inequality holds for SVO $\theta \geq 20°$ (covering prosocial and altruistic agents).
\end{itemize}
\end{proof}

\subsubsection{Phase W: Critique Defense (Revision)}

\textbf{This section addresses four critiques raised during peer review.}

\begin{definition}[Bounded Commitment]
\label{def:bounded}
An agent $i$ exhibits \textit{Bounded Commitment} if its commitment preserves a moral residue $\epsilon \cdot \sin(\theta_i)$ even under resource crisis ($R_t < R_{\text{crisis}}$):
\begin{equation}
\lambda_{\min}^{(i)} = \epsilon \cdot \sin(\theta_i), \quad \epsilon \in [0, 1]
\end{equation}
This ensures $\lambda_t^{(i)} \geq \lambda_{\min}^{(i)} > 0$ for all $\theta_i > 0$, distinguishing Bounded Commitment from the Maslow model ($\epsilon=0$) and the Martyr model ($\epsilon=1$).
\end{definition}

\begin{theorem}[Spillover Decomposition]
\label{thm:spillover}
In environments with agent interactions, the naive Average Treatment Effect (ATE) decomposes as:
\begin{equation}
\tau_{\text{naive}} = \tau_{\text{direct}} + \tau_{\text{spillover}}
\end{equation}
where $\tau_{\text{direct}} = \mathbb{E}[Y | d_i{=}1, e_i] - \mathbb{E}[Y | d_i{=}0, e_i]$ (exposure-controlled) and $\tau_{\text{spillover}} = \mathbb{E}[Y | d_i, e_i{=}\text{high}] - \mathbb{E}[Y | d_i, e_i{=}\text{low}]$ (treatment-controlled). The corrected AME reports $\tau_{\text{direct}}$ as the primary estimand.
\end{theorem}

\begin{itemize}
    \item \textbf{W1 --- SUTVA Analysis}: Explicit SUTVA violation measurement across PGG (no interference), Grid World (spatial interference), and Network PGG (network interference). Spillover accounts for $\approx$0 in PGG and up to 15\% of naive ATE in Grid World. Effect direction preserved in all cases (Fig. 75--78).
    \item \textbf{W2 --- Bounded Commitment Spectrum}: Four commitment models (Maslow $\to$ Sen $\to$ Bounded $\to$ Martyr) compared across 7 SVO conditions. Bounded ($\epsilon{=}0.3$) achieves Pareto-optimal survival-welfare balance. Moran Process confirms evolutionary stability (Fig. 79--80).
    \item \textbf{W3 --- Integrity-Constrained $U_{\text{meta}}$}: Four $U_{\text{meta}}$ variants (Naive, Bounded-kNN, KL-Divergence, Causal-Shapley) tested against Reward Pumping adversaries. Bounded-kNN preserves 97\% normal performance while resisting 30\% adversarial fraction. Computational overhead $<$3\% (Fig. 81--84).
    \item \textbf{W4 --- Cross-Domain Behavioral Transfer}: 4-dimensional Behavioral Fingerprint (crisis sensitivity, contribution elasticity, Gini dynamics, role stability) measured across PGG$\to$IPD, PGG$\to$Grid World, PGG$\to$Climate, IPD$\to$Grid World. Prosocial agents show highest cross-domain cosine similarity (Fig. 85--88).
\end{itemize}

\subsection{W5: SOTA Baseline Comparison (M-FOS / LOPT / LOLA)}
\label{app:sota}

\textbf{Motivation.} A critical question for NeurIPS reviewers is how Meta-Ranking compares to recent opponent-shaping and mechanism design algorithms. We benchmark against three state-of-the-art approaches:

\begin{itemize}
    \item \textbf{M-FOS} \citep{lu2022mfos}: Model-Free Opponent Shaping --- shapes opponents' learning trajectories via meta-learning. Requires O($N^2$) pairwise interactions.
    \item \textbf{LOPT} \citep{gemp2024lopt}: Learns optimal Pigovian taxes to internalize externalities. Requires a \textit{central regulatory agent}.
    \item \textbf{LOLA} \citep{foerster2018lola}: Learning with Opponent-Learning Awareness --- predicts opponents' gradient steps. Requires Hessian approximation, O($N^2$).
\end{itemize}

\textbf{Key advantages of Meta-Ranking:}
\begin{enumerate}
    \item \textbf{Decentralized}: No external regulator (unlike LOPT) or pairwise interaction (unlike M-FOS/LOLA). Each agent uses only local $R_t$ and own SVO $\theta$.
    \item \textbf{O(N) scaling}: Time per step grows linearly (0.44ms at N=1000) vs O($N^2$) for M-FOS (11.9ms) and LOLA (9.9ms).
    \item \textbf{Adversarial robustness}: At 30\% adversarial agents, Meta-Ranking maintains Coop=0.700 while M-FOS collapses to 0.000.
    \item \textbf{Scale invariance}: Cooperation rate remains 1.000 from N=20 to N=1000 while competitors degrade.
\end{enumerate}

\subsection{Philosophical Justification: From Sen's Commitment to Bounded Commitment}
\label{app:philosophy}

A potential critique is that our dynamic $\lambda_t$ mechanism, which reduces moral commitment during resource crises, departs from Sen's (1977) conception of commitment as fundamentally \textit{non-situational}---an unconditional willingness to sacrifice personal welfare for ethical principles.

\textbf{We acknowledge this departure and defend it on three grounds:}

\begin{enumerate}
    \item \textbf{Computational Survival Constraint.} Sen's framework was developed for human agents with biological survival guaranteed by institutional safety nets. Artificial agents in adversarial environments lack such guarantees. An agent that maintains unconditional commitment under resource crisis ($w < w_{\text{survival}}$) is \textit{eliminated}---and dead agents cannot commit to anything. Our Bounded Commitment (Definition~\ref{def:bounded}) preserves moral residue ($\lambda_{\min} = \epsilon \cdot \sin(\theta) > 0$) even in crisis, ensuring the agent \textit{survives to commit again}.

    \item \textbf{Empirical Vindication.} Our Moral Theory Tournament (P2, Fig.~53--54) demonstrates that unconditional Kantian Deontology agents achieve 0\% evolutionary survival against Byzantine populations, while Bounded Commitment achieves 99\%+ ESS. Situational flexibility is not moral weakness---it is the \textit{necessary condition for sustained ethical agency} in hostile environments.

    \item \textbf{Philosophical Precedent.} Sen himself noted that commitment ``drives a wedge between personal choice and personal welfare'' but did not prescribe \textit{self-destruction} as a consequence. Williams (1981) and Nussbaum (2001) similarly argue that moral agency requires the preservation of the agent's capacity for future moral action---the ``integrity constraint'' formalized in our W3 experiments.
\end{enumerate}

Thus, Bounded Commitment should be understood not as a \textit{betrayal} of Sen's theory but as its \textit{computational extension} to survival-constrained artificial agents, preserving the core insight that commitment is qualitatively distinct from sympathy while acknowledging the material conditions of agency.

\section{Broader Impact}
\label{app:impact}

\textbf{Positive.}
Our work provides a principled framework for designing AI agents that balance self-interest with social welfare. The policy simulations (Q6) demonstrate practical applicability to AI regulation and environmental governance.

\textbf{Risks.}
We acknowledge two risks:
(1) The ``weaponization'' of meta-ranking for sophisticated exploitation strategies, though our Byzantine robustness results (P6) suggest this is difficult.
(2) Over-reliance on simulation results for real-world policy, though our human pilot study (O8) begins bridging this gap.

\textbf{Limitations.}
Q1 (human experiments with $N{=}30$--60) remains future work requiring IRB approval and physical lab access. Our simulation results, while comprehensive, should be validated with human subjects.
